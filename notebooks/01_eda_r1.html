# Step 4: 5x5 grid — 5 random samples per emotion
import random
random.seed(42)   # fix seed for reproducibility

fig, axes = plt.subplots(5, 5, figsize=(12, 12))
fig.suptitle("FER-2013 Sample Images — 5 Emotions × 5 Samples",
            fontsize=14, fontweight='bold', y=1.01)

for row, emotion in enumerate(EMOTIONS):
    folder  = TRAIN_DIR + emotion + "/"
    files   = os.listdir(folder)
    samples = random.sample(files, min(5, len(files)))

    for col, fname in enumerate(samples):
        path = folder + fname
        img  = cv2.imread(path)

        if img is None:
            axes[row][col].axis("off")
            continue

        # Convert BGR → RGB for correct matplotlib display
        # OpenCV loads in Blue-Green-Red order
        # matplotlib expects Red-Green-Blue order
        img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

        axes[row][col].imshow(img_rgb, cmap='gray')
        axes[row][col].axis("off")

        # Add emotion label on leftmost column only
        if col == 0:
            axes[row][col].set_ylabel(emotion.upper(),
                                    fontsize=11, fontweight='bold', rotation=90, labelpad=10)
            axes[row][col].yaxis.set_label_position('left')

plt.tight_layout()
plt.savefig("eda_02_sample_grid.png", dpi=150, bbox_inches='tight')

sat_data = {e: [] for e in EMOTIONS}

    for fname in tqdm(os.listdir(folder)[:SAMPLE_N], desc=emotion, leave=False):

        sat_data[emotion].append(hsv[:,:,1].mean())   # S channel
        hue_data[emotion].append(float(hsv[:,:,0].var()))  # H variance

hog_data = {e: [] for e in EMOTIONS}

for emotion in EMOTIONS:
    folder = TRAIN_DIR + emotion + "/"
    for fname in tqdm(os.listdir(folder)[:SAMPLE_N], desc=emotion, leave=False):
        img = cv2.imread(folder + fname)
        if img is None: continue
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        gray = cv2.resize(gray, (48, 48))
        h = hog(gray, pixels_per_cell=(8,8), cells_per_block=(2,2))
        hog_data[emotion].append(float(np.mean(h ** 2)))

# What to look for:
# → surprise and angry should show clearly higher HOG energy
# → neutral should be lowest
# → If this separates → confirms F7 is a strong feature for Experiment B

edge_data = {e: [] for e in EMOTIONS}

for emotion in EMOTIONS:
    folder = TRAIN_DIR + emotion + "/"
    for fname in tqdm(os.listdir(folder)[:SAMPLE_N], desc=emotion, leave=False):
        img = cv2.imread(folder + fname)
        if img is None: continue
        gray  = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        edges = cv2.Canny(gray, 50, 150)
        edge_data[emotion].append(edges.sum() / edges.size)

# What to look for:
# → surprise should have highest edge density (wide eyes + open mouth)
# → neutral should be lowest
# → happy may be second highest (smile creates edges at mouth corners)

print("\n=== Feature Signal Summary ===")
print(f"{'Feature':<22} {'Max Class':<12} {'Min Class':<12} {'Ratio':>7}  Signal")
print("-" * 65)

datasets_named = [
    ("F1 Brightness Mean", brightness_data),
    ("F4 Saturation Mean", sat_data),
    ("F7 HOG Energy",      hog_data),
    ("F24 Edge Density",   edge_data),
]

for fname, data in datasets_named:
    means    = {e: np.mean(data[e]) for e in EMOTIONS}
    max_cls  = max(means, key=means.get)
    min_cls  = min(means, key=means.get)
    ratio    = means[max_cls] / (means[min_cls] + 1e-9)
    signal   = "Strong" if ratio > 1.3 else ("Moderate" if ratio > 1.1 else "Weak")
    print(f"  {fname:<20} {max_cls:<12} {min_cls:<12} {ratio:>6.2f}x  {signal}")

print("\nInterpretation:")
print("  Ratio > 1.3x  → feature clearly separates at least one emotion")
print("  Ratio < 1.1x  → feature adds mostly noise → consider dropping")
