import random
random.seed(42)   # fix seed for reproducibility

fig, axes = plt.subplots(5, 5, figsize=(12, 12))
fig.suptitle("FER-2013 Sample Images — 5 Emotions × 5 Samples",
            fontsize=14, fontweight='bold', y=1.01)

for row, emotion in enumerate(EMOTIONS):
    folder  = TRAIN_DIR + emotion + "/"
    files   = os.listdir(folder)
    samples = random.sample(files, min(5, len(files)))

    for col, fname in enumerate(samples):
        path = folder + fname
        img  = cv2.imread(path)

        if img is None:
            axes[row][col].axis("off")
            continue

        # Convert BGR → RGB for correct matplotlib display
        # OpenCV loads in Blue-Green-Red order
        # matplotlib expects Red-Green-Blue order
        img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

        axes[row][col].imshow(img_rgb, cmap='gray')
        axes[row][col].axis("off")

        # Add emotion label on leftmost column only
        if col == 0:
            axes[row][col].set_ylabel(emotion.upper(),
                                    fontsize=11, fontweight='bold', rotation=90, labelpad=10)
            axes[row][col].yaxis.set_label_position('left')

plt.tight_layout()
plt.savefig("eda_02_sample_grid.png", dpi=150, bbox_inches='tight')

upper_edge_data = {e: [] for e in EMOTIONS}
lower_edge_data = {e: [] for e in EMOTIONS}

for emotion in EMOTIONS:
    folder = TRAIN_DIR + emotion + "/"
    for fname in tqdm(os.listdir(folder)[:SAMPLE_N], desc=emotion, leave=False):
        img = cv2.imread(folder + fname)
        if img is None: continue
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        h, w = gray.shape
        upper = cv2.Canny(gray[:h//2,:], 50, 150)
        lower = cv2.Canny(gray[h//2:,:], 50, 150)
        upper_edge_data[emotion].append(upper.sum() / upper.size)
        lower_edge_data[emotion].append(lower.sum() / lower.size)

def boxplot_panel(ax, data_dict, title, ylabel):
    bp = ax.boxplot(
        [data_dict[e] for e in EMOTIONS],
        tick_labels=EMOTIONS,
        patch_artist=True,
        medianprops={'color':'black', 'linewidth':2},
        flierprops={'marker':'o', 'markersize':3, 'alpha':0.4}
    )
    for patch, color in zip(bp['boxes'], COLORS.values()):
        patch.set_facecolor(color)
        patch.set_alpha(0.75)
    ax.set_title(title, fontsize=11, fontweight='bold')
    ax.set_xlabel("Emotion")
    ax.set_ylabel(ylabel)

print("\n=== Feature Signal Summary ===")
print(f"  {'Feature':<26} {'Max Class':<12} {'Min Class':<12} {'Ratio':>7}  Signal  Decision")
print("-" * 85)

all_features = [
    ("F1  Brightness Mean",      brightness_data),
    ("F4  Saturation Mean",      sat_data),
    ("F24 Upper Edge Density",   upper_edge_data),
    ("F24 Lower Edge Density",   lower_edge_data),
]

for fname, data in all_features:
    means   = {e: np.mean(data[e]) for e in EMOTIONS}
    max_cls = max(means, key=means.get)
    min_cls = min(means, key=means.get)
    denom   = means[min_cls] + 1e-9
    ratio   = means[max_cls] / denom

    if ratio > 1.3:
        signal, decision = "Strong",  "KEEP"
    elif ratio > 1.1:
        signal, decision = "Moderate","KEEP"
    else:
        signal, decision = "Weak",    "REDESIGN/DROP"

    print(f"  {fname:<26} {max_cls:<12} {min_cls:<12} {ratio:>6.2f}x  {signal:<14} {decision}")
